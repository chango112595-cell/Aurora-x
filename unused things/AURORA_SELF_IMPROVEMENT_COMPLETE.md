def binary_search(arr: list, target: int) -> int:
    """Binary Search - O(log n) for sorted arrays"""
    left, right = 0, len(arr) - 1
    
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
