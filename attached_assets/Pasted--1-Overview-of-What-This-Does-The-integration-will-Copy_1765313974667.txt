üß© 1. Overview of What This Does

The integration will:

Copy the generated aurora_supervisor/ folder into your Aurora-X root directory.

Register the supervisor inside Aurora Nexus V3 so it can coordinate healers, workers, and phases 1-6.

Add startup hooks so every time Aurora-X boots, the supervisor launches automatically, resumes its learning context, and starts the healers.

Keep the whole thing reversible (you can disable or remove the supervisor cleanly if needed).

# --- Step 1: Move the supervisor into Aurora-X root ---------------------
mkdir -p /home/runner/Aurora-X/aurora_supervisor
cp -r aurora_supervisor/* /home/runner/Aurora-X/aurora_supervisor/

# --- Step 2: Create integration hook for Nexus V3 -----------------------
cat <<'PYCODE' > /home/runner/Aurora-X/aurora_nexus_v3/integrations/supervisor_integration.py
import os, sys, threading, time
from pathlib import Path
from importlib import import_module

AURORA_ROOT = Path(__file__).resolve().parents[2]
SUPERVISOR_PATH = AURORA_ROOT / "aurora_supervisor"

if str(SUPERVISOR_PATH) not in sys.path:
    sys.path.append(str(SUPERVISOR_PATH))

def start_supervisor():
    try:
        from supervisor_core import SupervisorCore
        core = SupervisorCore()
        t = threading.Thread(target=core.run, daemon=True)
        t.start()
        print("[Integration] SupervisorCore launched successfully.")
        return core
    except Exception as e:
        print("[Integration] Failed to start SupervisorCore:", e)
        return None

def attach_to_nexus_v3(nexus_instance):
    """
    Called during Aurora Nexus V3 boot sequence.
    Provides Nexus with SupervisorCore reference and hooks.
    """
    supervisor = start_supervisor()
    if supervisor:
        nexus_instance.supervisor = supervisor
        nexus_instance.log("SupervisorCore attached.")
PYCODE

# --- Step 3: Modify Nexus V3 startup to call the integration ------------
cat <<'PYCODE' >> /home/runner/Aurora-X/aurora_nexus_v3/main.py

# ===== Supervisor Integration Hook (Phase 4-6 Controller) =====
try:
    from aurora_nexus_v3.integrations.supervisor_integration import attach_to_nexus_v3
    attach_to_nexus_v3(nexus_instance)
except Exception as e:
    print("[Startup] Supervisor integration skipped or failed:", e)
# ===============================================================
PYCODE

# --- Step 4: Restart Aurora-X to activate Supervisor --------------------
cd /home/runner/Aurora-X
python3 aurora_nexus_v3/main.py
